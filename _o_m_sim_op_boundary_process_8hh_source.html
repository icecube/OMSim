305 inline void G4OpBoundaryProcess::ChooseReflection()
306 {
307  G4double rand = G4UniformRand();
308  if(rand < fProb_ss)
309  {
310  fStatus = SpikeReflection;
311  fFacetNormal = fGlobalNormal;
312  }
313  else if(rand < fProb_ss + fProb_sl)
314  {
315  fStatus = LobeReflection;
316  }
317  else if(rand < fProb_ss + fProb_sl + fProb_bs)
318  {
319  fStatus = BackScattering;
320  }
321  else
322  {
323  fStatus = LambertianReflection;
324  }
325 }

327 inline void G4OpBoundaryProcess::DoAbsorption()
328 {
329  fStatus = Absorption;
330 
331  if(G4BooleanRand(fEfficiency))
332  {
333  // EnergyDeposited =/= 0 means: photon has been detected
334  fStatus = Detection;
335  aParticleChange.ProposeLocalEnergyDeposit(fPhotonMomentum);
336  }
337  else
338  {
339  aParticleChange.ProposeLocalEnergyDeposit(0.0);
340  }
341 
342  fNewMomentum = fOldMomentum;
343  fNewPolarization = fOldPolarization;
344 
345  aParticleChange.ProposeTrackStatus(fStopAndKill);
346 }
347 
348 inline void G4OpBoundaryProcess::DoReflection()
349 {
350  if(fStatus == LambertianReflection)
351  {
352  fNewMomentum = G4LambertianRand(fGlobalNormal);
353  fFacetNormal = (fNewMomentum - fOldMomentum).unit();
354  }
355  else if(fFinish == ground)
356  {
357  fStatus = LobeReflection;
358  if(!fRealRIndexMPV || !fImagRIndexMPV)
359  {
360  fFacetNormal = GetFacetNormal(fOldMomentum, fGlobalNormal);
361  }
362  // else
363  // complex ref. index to be implemented
364  fNewMomentum =
365  fOldMomentum - (2. * fOldMomentum * fFacetNormal * fFacetNormal);
366  }
367  else
368  {
369  fStatus = SpikeReflection;
370  fFacetNormal = fGlobalNormal;
371  fNewMomentum =
372  fOldMomentum - (2. * fOldMomentum * fFacetNormal * fFacetNormal);
373  }
374  fNewPolarization =
375  -fOldPolarization + (2. * fOldPolarization * fFacetNormal * fFacetNormal);
376 }
