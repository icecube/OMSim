283 inline G4double OMSimG4Scintillation::bi_exp(G4double t, G4double tau1,
284  G4double tau2)
285 {
286  return std::exp(-1.0 * t / tau2) * (1 - std::exp(-1.0 * t / tau1)) / tau2 /
